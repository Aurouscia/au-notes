        private void MapLoad(int length, int height ,int blockSize,int margin)
        {
            Width = (length * blockSize + 2 * margin)+20;
            Height = (height * blockSize + 2 * margin)+55;
            List<Button> minelist = new List<Button>();
            Size size=new Size(blockSize,blockSize);
            Point point=new Point();
            Font ft = new Font("微软雅黑",7f);
            for(int i = 0; i < length; i++)
            {
                for(int j=0; j < height; j++)
                {
                    Button btn = new Button();
                    btn.Name = string.Format("{0},{1}", i, j);
                    btn.Text = string.Format("{0},{1}", i, j);
                    btn.Font = ft;
                    point.X = blockSize * i+margin;
                    point.Y = blockSize * j+margin;
                    btn.Size = size;
                    btn.Location = point;
                    btn.Click += BlockClicked;
                    btn.Image = Properties.Resources.X;
                    minelist.Add(btn);
                }
            }
            Controls.AddRange(minelist.ToArray());
            Text = Width.ToString();
        }

        private void BlockClicked(object sender, EventArgs e)
        {
            Text = ((Button)sender).Text;
        }
