在app\middleware中开启session
不支持原生获取方式
Session::set('键','值');
Session::get('键','默认值');
Session::all();
Session::has('');是否已赋值
Session::delete('')
Session::pull('')取值后删除
Session::clear();

Session::set('obj.user','Mr.Lee');
Session::has('obj.user');

Cookie::set('user','Mr.Lee');
Cookie::set('user','Mr.Lee',3600);过期时间
::get('')
::delete('')
::has('')
