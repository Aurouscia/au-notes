    private bool EndWithSaurus(string s)  
    {  
        if ((s.Length > 5) && (s.Substring(s.Length - 6).ToLower() == "saurus"))  
            return true;  
        else  
            return false;  
    } 
     List<string> subList = list.FindAll(EndWithSaurus);//传入了一个方法名   

列表可以存储任何类型的数据，<>里填类型
List<int>scoreList = new List<int>();
也可以匿名类型创建
var scoreList = new List<int>();
可以填初始值
var scoreList = new List<int>(){1,2,3};

scoreList.Add(12);
Console.WriteLine(scoreList[0]);通过索引访问数据
访问参数不可以超出范围

Insert(位置，数据)
会把后面的元素往后挤一位，
只能插在最后一位往前的
RemoveAt()
IndexOf()    正向寻找某元素的索引
LastIndexOf()    反向寻找
如果没有找到就返回-1
Sort()    从小到大的排序

构造函数里可以填一个初始大小
当位置不够用时，大小会自动扩大为原来的两倍
