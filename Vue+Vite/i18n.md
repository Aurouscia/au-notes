# i18n

## 1. 安装

```bash
npm install vue-i18n@latest
```

## 2. 定义 i18n 文本接口

把项目中所有文本字面量汇总为一个接口，确保类型安全：

```ts
// types/i18n.ts
export interface I18nMessages {
  home: {
    title: string
    subtitle: string
    welcome: string
    [key: string]: string
  }
  common: {
    confirm: string
    cancel: string
    save: string
    delete: string
    loading: string
    [key: string]: string
  }
  [key: string]: Record<string, unknown>
}

export type Locale = 'en' | 'zh'
```

## 3. 创建语言文件

为每种语言创建独立的翻译文件：

```ts

```

```ts

import type { I18nMessages } from '../types'
```

## 4. 创建 i18n 实例

```ts
// i18n/index.ts
import { createI18n } from 'vue-i18n'
import zh from './locales/zh'
import en from './locales/en'

const messages = { en, zh }

export const i18n = createI18n({
  legacy: false,        // 使用 Composition API 模式
  locale: 'zh',         // 默认语言
  fallbackLocale: 'en', // 回退语言
  messages,             // 翻译文本
  globalInjection: true // 全局注入 $t 函数
})

export default i18n
```

### createI18n 配置选项说明

| 选项 | 类型 | 说明 |
|------|------|------|
| `legacy` | `boolean` | 是否使用 Vue 2 兼容模式，建议设为 `false` |
| `locale` | `string` | 当前默认语言 |
| `fallbackLocale` | `string` | 当前语言缺少翻译时使用的回退语言 |
| `messages` | `object` | 语言包对象 |
| `globalInjection` | `boolean` | 是否全局注入 `$t` 等 API |
| `silentTranslationWarn` | `boolean` | 是否静默翻译警告 |

## 5. 在 main.ts 中注册

```ts
// main.ts
import { createApp } from 'vue'
import App from './App.vue'
import i18n from './i18n/index'

const app = createApp(App)

app.use(i18n)
app.mount('#app')
```

## 6. 在模板中使用

### 6.1 使用 `$t()` 函数

```vue
<template>
  <div class="home">
    <h1>{{ $t('home.title') }}</h1>
    <p>{{ $t('home.subtitle') }}</p>
    <button>{{ $t('common.confirm') }}</button>
  </div>
</template>
```

### 6.2 使用插值表达式（带参数）

```ts
// 语言文件
const messages = {
  home: {
    greeting: '你好，{name}！你有 {count} 条新消息。'
  }
}
```

```vue
<template>
  <p>{{ $t('home.greeting', { name: '张三', count: 5 }) }}</p>
  <!-- 输出: 你好，张三！你有 5 条新消息。 -->
</template>
```

### 6.3 复数形式

#### 基础用法（两种形式）

```ts
// 语言文件
const messages = {
  cart: {
    itemCount: '购物车中没有商品 | 购物车中有 {n} 件商品'
  }
}
```

```vue
<template>
  <p>{{ $tc('cart.itemCount', 0) }}</p>   <!-- 购物车中没有商品 -->
  <p>{{ $tc('cart.itemCount', 3) }}</p>   <!-- 购物车中有 3 件商品 -->
</template>
```

#### 多种形式（0、1、1+ 三种情况）

使用管道符 `|` 分隔多种形式，vue-i18n 会根据数量自动选择：

```ts
// 语言文件
const messages = {
  cart: {
    // 格式: 0项 | 1项 | 多项
    itemCount: '购物车是空的 | 购物车中有 1 件商品 | 购物车中有 {n} 件商品'
  },
  user: {
    // 格式: 0封 | 1封 | 多封
    newMessages: '你没有新消息 | 你有 1 条新消息 | 你有 {n} 条新消息'
  }
}
```

```vue
<template>
  <!-- 0 件 -->
  <p>{{ $tc('cart.itemCount', 0) }}</p>   <!-- 购物车是空的 -->
  
  <!-- 1 件 -->
  <p>{{ $tc('cart.itemCount', 1) }}</p>   <!-- 购物车中有 1 件商品 -->
  
  <!-- 多件 -->
  <p>{{ $tc('cart.itemCount', 5) }}</p>   <!-- 购物车中有 5 件商品 -->
</template>
```

#### 自定义复数规则（复杂场景）

如果默认的复数规则不满足需求，可以自定义复数规则：

```ts
// i18n/index.ts
const i18n = createI18n({
  legacy: false,
  locale: 'zh',
  pluralRules: {
    // 自定义中文复数规则
    zh: (choice: number, choicesLength: number) => {
      // choice: 传入的数字
      // choicesLength: 可选形式的数量
      
      if (choice === 0) return 0  // 第1项（0个）
      if (choice === 1) return 1  // 第2项（1个）
      return 2                     // 第3项（多个）
    },
    // 自定义英文复数规则
    en: (choice: number, choicesLength: number) => {
      if (choice === 0) return 0
      if (choice === 1) return 1
      return 2
    }
  },
  messages
})
```

#### 使用对象形式定义复数（推荐）

更清晰的对象写法，支持任意数量的形式：

```ts
// 语言文件
const messages = {
  cart: {
    itemCount: {
      zero: '购物车是空的',
      one: '购物车中有 1 件商品',
      other: '购物车中有 {n} 件商品'
    }
  }
}
```

```vue
<script setup lang="ts">
import { useI18n } from 'vue-i18n'

const { t } = useI18n()

function getItemCountText(count: number) {
  if (count === 0) return t('cart.itemCount.zero')
  if (count === 1) return t('cart.itemCount.one')
  return t('cart.itemCount.other', { n: count })
}
</script>

<template>
  <p>{{ getItemCountText(0) }}</p>   <!-- 购物车是空的 -->
  <p>{{ getItemCountText(1) }}</p>   <!-- 购物车中有 1 件商品 -->
  <p>{{ getItemCountText(5) }}</p>   <!-- 购物车中有 5 件商品 -->
</template>
```

## 7. 在组合式 API 中使用

### 7.1 基本用法

```vue
<script setup lang="ts">
import { useI18n } from 'vue-i18n'

const { t, locale } = useI18n()

// 使用 t 函数翻译
const title = t('home.title')
</script>

<template>
  <h1>{{ title }}</h1>
</template>
```

### 7.2 切换语言

```vue
<script setup lang="ts">
import { useI18n } from 'vue-i18n'
import type { Locale } from './types'

const { locale } = useI18n()

function switchLanguage(lang: Locale) {
  locale.value = lang
  // 可选：保存到 localStorage
  localStorage.setItem('locale', lang)
}

// 页面加载时恢复语言设置
const savedLocale = localStorage.getItem('locale') as Locale
if (savedLocale) {
  locale.value = savedLocale
}
</script>

<template>
  <div class="language-switcher">
    <button @click="switchLanguage('zh')">中文</button>
    <button @click="switchLanguage('en')">English</button>
  </div>
</template>
```

### 7.3 useI18n 返回值详解

```ts
const {
  t,           // 翻译函数
  tc,          // 复数翻译函数
  te,          // 检查键是否存在
  tm,          // 获取消息对象
  locale,      // 当前语言（ref）
  fallbackLocale, // 回退语言
  availableLocales, // 可用语言列表
  messages     // 所有消息
} = useI18n()
```

## 8. 进阶用法

### 8.1 日期时间格式化

```ts
// i18n/index.ts
const i18n = createI18n({
  legacy: false,
  locale: 'zh',
  datetimeFormats: {
    zh: {
      short: { year: 'numeric', month: 'short', day: 'numeric' },
      long: { year: 'numeric', month: 'long', day: 'numeric', weekday: 'long' }
    },
    en: {
      short: { year: 'numeric', month: 'short', day: 'numeric' },
      long: { year: 'numeric', month: 'long', day: 'numeric', weekday: 'long' }
    }
  }
})
```

```vue
<template>
  <p>{{ $d(new Date(), 'short') }}</p>
  <p>{{ $d(new Date(), 'long') }}</p>
</template>
```

### 8.2 数字格式化

```ts
// i18n/index.ts
const i18n = createI18n({
  legacy: false,
  locale: 'zh',
  numberFormats: {
    zh: {
      currency: { style: 'currency', currency: 'CNY' },
      percent: { style: 'percent', minimumFractionDigits: 2 }

    en: {
      currency: { style: 'currency', currency: 'USD' },

```

```vue
<template>
  <p>{{ $n(1234.5, 'currency') }}</p>
  <p>{{ $n(0.85, 'percent') }}</p>
</template>
```

### 8.3 异步加载语言包

```ts
// i18n/index.ts
import { createI18n } from 'vue-i18n'

const i18n = createI18n({
  legacy: false,
  locale: 'zh',
  fallbackLocale: 'en',
  messages: {}
})

// 异步加载语言包
export async function loadLocaleMessages(locale: string) {
  const messages = await import(`./locales/${locale}.ts`)
  i18n.global.setLocaleMessage(locale, messages.default)
  i18n.global.locale.value = locale
}

export default i18n
```

### 8.4 自定义指令

```ts
// 自定义 v-t 指令
import { createI18n } from 'vue-i18n'

const i18n = createI18n({
  legacy: false,
  locale: 'zh'
})

// 注册自定义指令
app.directive('t', {
  mounted(el, binding) {
    el.textContent = i18n.global.t(binding.value)
  },
  updated(el, binding) {
    el.textContent = i18n.global.t(binding.value)
  }
})
```

```vue
<template>
  <span v-t="'home.title'"></span>
</template>
```

## 9. 最佳实践

### 9.1 项目结构建议

```
src/
├── i18n/
│   ├── index.ts          # i18n 实例创建
│   ├── locales/
│   │   ├── zh.ts         # 中文翻译
│   │   ├── en.ts         # 英文翻译
│   │   └── ja.ts         # 日文翻译
│   └── types.ts          # 类型定义
├── components/
│   └── LanguageSwitcher.vue
└── main.ts
```

### 9.2 键名命名规范

- 使用点号分隔的层级结构：`模块.页面.元素`
- 例如：`user.profile.saveButton`、`common.errors.network`
- 保持键名简洁且具有描述性

### 9.3 提取工具

使用 `vue-i18n-extract` 工具提取和检查翻译键：

```bash
# 安装
npm install -D vue-i18n-extract

# 检查缺失的翻译
npx vue-i18n-extract report --vueFiles './src/**/*.vue' --languageFiles './src/i18n/locales/*.ts'
```

## 10. 常见问题

### 10.1 切换语言后组件不更新

确保使用 `ref` 或 `computed` 包裹翻译内容，或直接在模板中使用 `$t`。

### 10.2 TypeScript 类型错误

在 `env.d.ts` 中添加类型声明：

```ts
// env.d.ts
declare module 'vue' {
  export interface ComponentCustomProperties {
    $t: typeof import('vue-i18n').t
    $tc: typeof import('vue-i18n').tc
    $d: typeof import('vue-i18n').d
    $n: typeof import('vue-i18n').n
  }
}

export {}
```

### 10.3 Vite 配置优化

```ts
// vite.config.ts
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import VueI18nPlugin from '@intlify/unplugin-vue-i18n/vite'

export default defineConfig({
  plugins: [
    vue(),
    VueI18nPlugin({
      include: ['./src/i18n/locales/**']
    })
  ]
})
```
