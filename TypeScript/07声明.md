# declare关键字
ts中，declare关键字用于声明类型信息而不提供具体实现，告诉编译器"这个变量/函数/类存在于运行时，但我现在不定义它"。

## 示例
### 告诉 TS：运行时全局会有一个 myLib 变量
```ts
declare const myLib: {
  version: string;
  doSomething(): void;
};

// 编写时不会报错，尽管没定义实现（myLib可能在别处，例如cdn，添加到全局了）
console.log(myLib.version);  // ✅
```

### 为 JavaScript 代码提供类型
```ts
// utils.js (纯 JS，无类型)
export function add(a, b) { return a + b; }

// utils.d.ts (类型声明文件)
export function add(a: number, b: number): number;
```

**错误示例**：`.d.ts` 文件中不需要 `declare export` 这种写法
```ts
// ❌ 错误：declare export 不是正确的语法
declare export function add(a: number, b: number): number;

// ✅ 正确：直接 export 即可（在 .d.ts 文件中 declare 是可选的）
export function add(a: number, b: number): number;

// ✅ 也正确：显式使用 declare 后再 export
declare function add(a: number, b: number): number;
export { add };
```

### 为已有的模块提供/扩展类型
```ts
// types.d.ts
declare module 'legacy-lib' {
  // 这个库没有 @types 包，自己补充
  export function oldFunction(): void;
  export const VERSION: string;
}

// 使用时
import { oldFunction } from 'legacy-lib';  // ✅ 不再报错"找不到模块"
```

### 声明文件中的顶层导出
```ts
// index.d.ts
declare namespace MyLib {
  export interface Config {
    timeout: number;
  }
  
  export function configure(options: Config): void;
}

// 导出整个命名空间
export = MyLib;
export as namespace MyLib;  // 同时作为全局变量可用
```

### 声明环境变量
```ts
// env.d.ts
declare namespace NodeJS {
  interface ProcessEnv {
    readonly API_URL: string;
    readonly NODE_ENV: 'development' | 'production';
  }
}

// 使用时
process.env.API_URL;  // ✅ 类型是 string，且有自动补全
```

**为什么这里不需要 `export`？**
- `NodeJS` 是 Node.js 类型定义中**已经存在的全局 namespace**
- 这里做的是**声明合并（Declaration Merging）**，向已有 namespace 添加接口
- 扩展已有 namespace 时，成员不需要 `export` 也能被合并进去

**等价的现代写法（推荐）**
```ts
// env.d.ts
// 使用 declare global 更明确地表达"扩展全局"
declare global {
  namespace NodeJS {
    interface ProcessEnv {
      readonly API_URL: string;
      readonly NODE_ENV: 'development' | 'production';
    }
  }
}

export {};  // 加上这行让文件变成模块，避免污染全局
```

### 扩展全局类型
```ts
declare global {
  interface Window {
    __REDUX_DEVTOOLS_EXTENSION__: any;
  }
}
```

## 可见性
对于`declare namespace ABC`
- 如果声明文件没有 export 或 import 语句（即"脚本"模式），那么`ABC`会成为全局命名空间
- 如果声明文件包含 export 或 import，它就是一个模块，此时 declare namespace 不会自动暴露到全局，解决方法：
    - 在需要使用的文件中显式导入
    - 使用 declare global 包裹来强制提升到全局
        ```ts
        declare global {
            namespace ABC {
                export const foo: string;
            }
        }
        ```

# export 在 ts 中的扩展语法
## export as namespace MyLib
可使 MyLib 在全局可访问（用于cdn场景）

## export = xxx
对应于 commonJS 中的 `module.exports = xxx`，将整个模块作为单个对象导出。
```ts
// fs.d.ts
declare module 'fs' {
  function readFile(path: string): void;
  // ... 很多函数
  export = fs;  // 整体导出
}
```

## export as namespace
用于 **UMD 模块**，让模块在通过 `<script>` 标签引入时也能作为全局变量访问。
```ts
// lodash.d.ts (UMD 格式)
export = _;           // CommonJS 方式导出
export as namespace _; // 同时作为全局变量 _ 可用

// 使用时：
// 作为模块：import * as _ from 'lodash';
// 作为全局：_.debounce(...) （CDN 引入时）
```

## 三者的关系和区别

| 语法 | 模块系统 | 能否与 `export =` 共存 | 使用场景 |
|------|----------|----------------------|----------|
| `export = xxx` | CommonJS | - | Node.js 库 |
| `export as namespace X` | UMD | ✅ 可以 | 需要同时支持模块和全局的库 |
| `export const xxx` | ESM | ❌ 不可以 | 现代 ES 模块 |

### 组合使用示例（UMD 格式）
```ts
// my-lib.d.ts
export function foo(): void;
export const version: string;

export = myLib;              // CommonJS 导出
export as namespace myLib;   // 全局变量也可用

declare namespace myLib {
  export { foo, version };
}
```

### 错误示例
```ts
// ❌ 错误：export = 和 ESM export 不能同时出现
export = myLib;
export const foo = 1;  // 语法错误！
```

# namespace

## namespace 内部有无 export 的区别

在 `namespace` 内部，只有标记为 `export` 的成员才能被外部访问。

```ts
namespace MyLib {
  // 有 export：外部可访问
  export function publicFn() { }
  export const publicVar = 1;
  
  // 无 export：仅在 namespace 内部可见
  function privateFn() { }
  const privateVar = 2;
  
  export function usePrivate() {
    // 内部可以访问未导出的成员
    privateFn();
    return privateVar;
  }
}

// 使用：
MyLib.publicFn();      // ✅ 可以访问
MyLib.usePrivate();    // ✅ 可以访问
// MyLib.privateFn();  // ❌ 错误：属性 'privateFn' 不存在
```

## namespace 不再推荐使用的原因

### 1. ES 模块是更好的选择
- **namespace** 是 TypeScript 特有的组织代码方式，编译后变成全局对象
- **ES 模块** 是 JavaScript 标准，有更好的工具支持和生态兼容性

### 2. 全局命名空间污染
```ts
// ❌ 使用 namespace 容易造成全局污染
namespace App {
  export function init() {}
}
// 编译后：var App = { init: ... }，挂在全局

// ✅ 使用 ES 模块更隔离
// app.ts
export function init() {}
// 使用时：import { init } from './app';
```

### 3. 模块解析更复杂
使用 namespace 时，需要依赖 `<reference path="..." />` 或确保文件加载顺序，而 ES 模块的依赖关系更清晰。

### 4. 现代项目的推荐做法
| 场景 | 推荐方式 |
|------|----------|
| 新项目 | 使用 ES 模块 (`import`/`export`) |
| 类型组织 | 使用 `interface`、`type` 或 `.d.ts` 文件 |
| 需要全局类型 | 使用 `declare global` |
| 遗留项目维护 | 可继续使用 namespace |

### 何时还可以使用 namespace
1. **描述外部库的类型**（如 jQuery 的 `$`）
2. **内部工具类型分组**（在 `.d.ts` 文件中）
3. **与旧代码兼容**

## `declare namespace` vs `namespace` 的区别

| 特性 | `namespace` | `declare namespace` |
|------|-------------|---------------------|
| 编译输出 | 生成 JavaScript 代码（对象） | 不生成代码，仅类型声明 |
| 使用场景 | 实现代码组织 | 为已有代码提供类型 |
| 文件类型 | `.ts` 文件 | `.d.ts` 文件 |

### `namespace` 编译后变成对象
```ts
// 源码
namespace MyLib {
  export function foo() { return "hello"; }
  export const version = "1.0.0";
}

// 编译后（简化）
var MyLib = {
  foo: function() { return "hello"; },
  version: "1.0.0"
};
```

### `declare namespace` 不产生代码
```ts
// types.d.ts
declare namespace ExternalLib {
  export function init(): void;
}

// 编译后：不产生任何 JavaScript 代码
// 仅告诉 TypeScript 编译器：运行时存在 ExternalLib 对象
```

# 为什么 DefinitelyTyped 拒绝 declare module 'xxx'

## 原因
在 DefinitelyTyped 中，类型定义包的目录名（如 `@types/lodash` 中的 `lodash`）**已经被 TypeScript 自动识别为模块名**。如果在 `index.d.ts` 中再写 `declare module 'lodash'`，会导致：

1. **重复声明**：模块被定义两次
2. **路径嵌套错误**：使用时会出现 `lodash/lodash` 这样的错误嵌套路径

## 示例说明

### ❌ 错误写法（在 DefinitelyTyped 中）
```ts
// types/foo/index.d.ts
declare module 'foo' {  // ❌ 错误：目录名已经是 'foo'，这里再 declare 会造成嵌套
  export function bar(): void;
}
```
使用时会出现错误路径：
```ts
import { bar } from 'foo/foo';  // 错误地变成了 foo/foo
```

### ✅ 正确写法
```ts
// types/foo/index.d.ts
// 直接写类型定义，不要包裹 declare module
export function bar(): void;
export const VERSION: string;
```

## 何时可以使用 declare module
在以下场景中使用 `declare module` 是合理的：

1. **为无类型的第三方库补充类型**（不是发布到 @types 的包）
   ```ts
   // 项目内的 types.d.ts
   declare module 'untyped-lib' {
     export function doSomething(): void;
   }
   ```

2. **扩展已有模块的类型**（模块扩充/插件类型）
   ```ts
   // 为 vue 添加自定义属性
   declare module 'vue' {
     interface ComponentCustomProperties {
       $myProperty: string;
     }
   }
   ```

3. **声明通配符模块**（资源文件）
   ```ts
   declare module '*.png' {
     const src: string;
     export default src;
   }
   ```
