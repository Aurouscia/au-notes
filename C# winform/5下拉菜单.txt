        public Dictionary<string, string> locs;

        private void Form_Load(object sender, EventArgs e)
        {
            locs = new Dictionary<string, string>();
            locs.Add("武汉市", "汉阳区，江岸区，武昌区");
            locs.Add("南京市", "鼓楼区，雨花台区，江宁区");
            foreach (var item in locs.Keys)
            {
                comboBox1.Items.Add(item.ToString());
            }
        }

        private void comboBox1_SelectedIndexChanged(object sender, EventArgs e)
        {
            【comboBox2.Items.Clear();】
            string selectedText = comboBox1.SelectedItem.ToString();
            string[] secondaryTexts = locs[selectedText].Split('，');
            comboBox2.Items.AddRange(secondaryTexts);
            comboBox2.SelectedIndex = 0;
        }
